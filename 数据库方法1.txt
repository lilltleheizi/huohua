    3： 查询语句优化
     1：索引列上 避免使用not 遇到”NOT,他就会停止使用索引转而执行全表扫描.
	 2: 避免在索引列上使用计算,优化器将不使用索引而使用全表扫描
	 3: 避免在索引列上使用IS NULL和IS NOT NULL 
	 4: 使用通配符%的情况下Oracle可能会停用该索引
	 5: 避免改变索引列的类型  字段类型是字符类型 对应的检索条件表达式应该是 字符类型的表达式  如果不是同一类型 会影响到索引

    //更新clob字段	 
declare
  v_sql clob;
  begin
  v_sql := rpad('*', 4021, '*') || rpad('*', 4000, '*');  
  update risk.t_basis_annual_risk_list set  risk_desc=v_sql ;
  commit;
  end;
